function mayorque (x, y){
	var cont = 0;
	for(var i=0; i < x.length; i++){
		if(x[i] > y){
			cont++;
		}
	}
	console.log("Hay " + cont + " valores mayores que Y");
}

function maxminavg (x){
	var max = 0;
	var min = 0;
	var avg = 0;
	for(var i=0; i < x.length; i++){
		if(x[i] > max){
			max = x[i];
		}
		if(x[i] < min){
			min = x[i];
		}
		avg+= x[i];
	}
	avg /= x.length;
	console.log("El mayor es " + max + ", el mÃ­nimo es " + min + " y el promedio es " + avg + ".");
}

function deleteinarray(x, y, z){
	var temp = [];
	for(var i=0; i < x.length; i++){
		if(i < y || i > z ){
			temp.push(x[i]);
		}
	}
	return temp;
}
